■ログインユーザーのみに操作を許可する例
alter policy "Individuals can create titles."
on "public"."manga_title"
to public
with check (
  (auth.uid() = user_id)
);

alter policy "Individuals can delete their own titles."
on "public"."manga_title"
to public
using (
  (auth.uid() = user_id)
);

alter policy "Individuals can update their own titles."
on "public"."manga_title"
to public
using (
  (auth.uid() = user_id)
);

alter policy "Individuals can view their own titles."
on "public"."manga_title"
to public
using (
  (auth.uid() = user_id)
);
