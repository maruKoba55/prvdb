DROP TABLE booktype_master;
CREATE TABLE booktype_master (
  booktype_cd         CHAR(03)     PRIMARY KEY
 ,booktype            TEXT
 ,selectable          BOOLEAN
);


üĪrole_master
DROP TABLE role_master;
CREATE TABLE role_master (
  role_cd             CHAR(03)     PRIMARY KEY
 ,role_name           TEXT
 ,selectable          BOOLEAN
);


üĪbooks
DROP TABLE books;
CREATE TABLE books (
  book_id             INTEGER      GENERATED BY DEFAULT AS IDENTITY
                                   (START WITH 10001)
                                   PRIMARY KEY
 ,isbn10              VARCHAR(10)
 ,isbn13              VARCHAR(13)
 ,c_cd                VARCHAR(05)
 ,ndc                 VARCHAR(10)
 ,title               TEXT         NOT NULL
 ,original_title      TEXT
 ,colophon            TEXT
 ,publisher           TEXT         NOT NULL
 ,publish_series      TEXT
 ,publish_series_no   TEXT
 ,first_publish_year  INTEGER      NOT NULL
 ,remarks             TEXT
 ,comic_f             BOOLEAN
 ,image_url           TEXT
 ,UNIQUE (title, publisher, first_publish_year)
);


üĪbook_possess
DROP TABLE book_possess;
CREATE TABLE book_possess (
  book_possess_id     INTEGER      GENERATED BY DEFAULT AS IDENTITY
                                   PRIMARY KEY
 ,book_id             INTEGER      REFERENCES books (book_id)
                                   ON DELETE CASCADE
 ,booktype_cd         CHAR(03)     REFERENCES booktype_master (booktype_cd)
 ,get_date            DATE         NOT NULL
 ,dispose_date        DATE
 ,image_url           TEXT
 ,remarks             TEXT
 ,UNIQUE (book_id, booktype_cd, get_date)
);


üĪbook_role
DROP TABLE book_role;
CREATE TABLE book_role (
  id                  INTEGER      GENERATED BY DEFAULT AS IDENTITY
                                   PRIMARY KEY
 ,book_id             INTEGER      REFERENCES books (book_id)
                                   ON DELETE CASCADE
 ,role_cd             CHAR(03)     REFERENCES role_master (role_cd)
 ,role_order          INTEGER
 ,person_name         TEXT
 ,remarks             TEXT
 ,UNIQUE (book_id, role_cd, person_name)
);


üĪbook_note
DROP TABLE book_note;
CREATE TABLE book_note (
  id                  INTEGER      GENERATED BY DEFAULT AS IDENTITY
                                   PRIMARY KEY
 ,book_id             INTEGER      REFERENCES books (book_id)
 ,read_st_date        DATE
 ,read_ed_date        DATE
 ,note                TEXT
 ,UNIQUE (book_id, read_st_date)
);
